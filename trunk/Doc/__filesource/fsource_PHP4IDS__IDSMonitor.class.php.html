  116.     function &Run()

  121.             foreach ($this->mRequest as $key => $value)

  129.     /**
  130.      * Iterates through given data and delegates it
  131.      * to IDSMonitor::Detect() in order to check for malicious
  132.      * appearing fragments
  133.      *
  134.      * @access private

  136.      * @param mixed $value

  138.      */
  139.     function Iterate($key, $value, &$report)
  140.     {
  141.         if (!is_array($value))
  142.         {
  143.             if (is_string($value))
  144.             {
  145.                 $filters=$this->Detect($key, $value);
  146.                 if (!is_null($filters))
  147.                 {
  148.                     $event=&new IDSEvent($key, $value, $filters);
  149.                     $report->AddEvent($event);
  150.                 }
  151.             }
  152.         }
  153.         else
  154.         {
  155.             foreach ($value as $subKey => $subValue)
  156.             {
  157.                 $this->Iterate($key . '.' . $subKey, $subValue, $report);
  158.             }
  159.         }
  160.     }

  162.     /**
  163.      * Checks whether given value matches any of the supplied
  164.      * filter patterns
  165.      *
  166.      * @access    private
  167.      * @param mixed $key 
  168.      * @param mixed $value 
  169.      * @return    array 
  170.      */
  171.     function Detect($key, $value)
  172.     {
  173.         if (preg_match('/[^\w\s\/]+/ims', $value) && !empty($value))
  174.         {
  175.             if (in_array($key, $this->mExceptions, true)) return null;
  176.  
  177.             $value=IDSConverter::Convert($value);
  178.             $value=get_magic_quotes_gpc() ? stripslashes($value) : $value;
  179.             $key=($this->mScanKeys) ? IDSConverter::Convert($key) : $key;
  180.  
  181.             $filters=array();
  182.             $filterSet=$this->mStorage->FilterSet();
  183.             for ($i=0; $i<count($filterSet); $i++)
  184.             {
  185.                 $filter=&$filterSet[$i];
  186.                 if (is_array($this->mTags))
  187.                 {
  188.                     if (array_intersect($this->mTags, $filter->Tags()))
  189.                     {
  190.                         if ($this->Match($key, $value, $filter)) $filters[]=&$filter;
  191.                     }
  192.                 }
  193.                 else
  194.                 {
  195.                     if ($this->Match($key, $value, $filter)) $filters[]=&$filter;
  196.                 }
  197.             }
  198.             return empty($filters) ? null : $filters;
  199.         }
  200.     }

  214.             if ($filter->Match($key)) return true;
  215.         }

  217.     }
